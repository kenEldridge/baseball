from os import listdir
from os.path import isfile, join
path = 'baseballdatabank-master/core/'
files = [f for f in listdir(path) if isfile(join(path, f)) and f[len(f) - 3:] == 'csv']
print(files)

dfs = {}
for f in files:
    dfs[f] = pd.read_csv(path + f)
    print(f)
    print(dfs[f].columns)
